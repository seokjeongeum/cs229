\begin{eqnarray*}
p(y; \eta)=b(y)\cdot \exp(\eta\cdot T(y)-a(\eta))=\frac{e^{-\lambda}\lambda^y}{y!}\\

\end{eqnarray*}

b(y)=\frac{1}{y!}\\
\eta=\log(\lambda)\\
T(y)=y\\
a(\eta)=e^{\eta}
